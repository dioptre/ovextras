SET(TEST_NAME PLDA-OneVsOne-HT-Test)
SET(SCENARIO_TO_TEST "${TEST_NAME}.xml")

ADD_TEST(run_${TEST_NAME} "$ENV{OV_BINARY_PATH}/openvibe-designer.${EXT}" ${OS_FLAGS} "--no-gui" "--play-fast" ${SCENARIO_TO_TEST})
